1、如果界面中只定义一个列表，使用标准列表实现即可。也可以使用gtiles-tasty-table指令完成列表呈现。
2、如果界面中需要做多列表展现，请使用gtiles-tasty-table指令完成列表呈现
指令说明：
<gtiles-tasty-table options="standardOption"></gtiles-tasty-table>
gtilesTastyTable模块引用
1、在angularjs模块中注册'ngTasty','gtilesTasty'两个模块。
2、在UIState中引用js文件，asserts/js/plugin/gtilesTasty/gtiles-tasty.js
3、在ctrl中定义gtiles-tasty-table的options，参数说明如下：
$scope.standardOption = {
"colDef":{
"isSelectAll" : true,//是否全选 必填
//列表列定义
"columns" : [{
"property":"roleId",//属性 必填
"desc":"角色名称",//描述 显示时必填
"primaryKey":true,//是否为主键 全选时用，不必填
"isDisplay":false//是否显示 默认显示
},{
"property":"roleName",
"desc":"角色名称",
"style":{ //列样式
"width":"70%"
}
},{
"property":"roleCode",
"desc":"角色编码"
}],
"opts":{
"onDelete":function(id){
console.log("onDelete..."+id);
}
}
},
// 列表加载方法 必填
"loadFun":function(params, paramsObj) {
return uitabtastyService.getRole(params, paramsObj);
},
//列表初始化对象
"init":{"sortBy" : "status","sortOrder" : "desc"}
};
 4、列表重新加载方法：$scope.standardOption.reloadData();
5、获取选择列表项方法：$scope.standardOption.datagrid.selectionIds
